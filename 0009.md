# 0009 · 지원/약속 매커니즘 정합성 보강

**파일**: `supabase/migrations/20260525120009_appointment_consistency.sql`
**카테고리**: 정합성
**의존성**: 0008

## 목적

N:M 구조에서 새로 노출된 정합성 결함 5가지를 한 번에 정리. 동일 펫 이중 약속 차단, 보호자 자기 지원 차단, 비활성 펫 게시글 지원 차단, 분양 완료 시 보호자 전원 정리, owner self-remove 차단.

## 핵심 변경

| 구분 | 내용 |
|---|---|
| [A] 펫 점유 | 펫 1마리당 활성 약속(scheduled) 1개 규칙 — applications_on_accept 사전검사 + appointments BEFORE INSERT 백스톱 2중 방어 |
| [B] 보호자 자기지원 | 신청자가 게시글 펫의 보호자이면 차단 |
| [C] 비활성 펫 포함 게시글 | 신규 지원 차단 |
| [D] 분양 완료 시맨틱 | 기존 보호자 전원 정리 → 입양자만 owner (0008의 강등 방식을 개선) |
| [E] owner self-remove 차단 | app.uid() = old.user_id 이면 BEFORE DELETE 트리거로 거절 |

## 전체 SQL

```sql
-- ============================================================================
-- PawMate · 0009 · 지원/약속 매커니즘 정합성 보강
-- ----------------------------------------------------------------------------
-- 새 N:M 구조에서 새로 노출된 결함 + 곁가지로 발견된 정합성 문제를 한 번에 정리.
--
-- [A] 동일 펫 이중 약속(현실 충돌) 방어
--   문제: 보호자 A·B 가 같은 펫으로 각자 글을 올리면, 양쪽에서 수락 시 펫이 동시에
--         두 개의 'scheduled' 약속에 잡혀 현실에서 꼬임. DB 자체는 허용해버림.
--   해결: "펫 1마리당 활성 약속(scheduled) 1개" 규칙을 트리거로 강제.
--         applications 수락 단계의 사전 검사(명확한 에러) + appointments INSERT
--         단계의 최종 백스톱 트리거(어떤 경로로 들어와도 차단) 2중 방어.
--
-- [B] 보호자 본인이 자기 게시글에 지원하는 케이스 차단
--   문제: 같은 펫의 공동보호자(B)가 owner(A)의 그 펫 게시글에 지원할 수 있었음.
--         "내 펫인데 내가 지원" 은 의미상 모순이고 평가 시스템도 꼬임.
--
-- [C] 비활성(deceased/deleted/transferred) 펫이 포함된 게시글 신규 지원 차단
--   문제: 글 작성 후 펫 상태가 비활성으로 바뀌면 그 글 지원이 의미 없어짐.
--
-- [D] 분양(give_away) 완료 시 보호자 처리 방식 변경
--   문제: 0008 은 기존 owner 만 co_guardian 으로 강등 — 다른 co_guardian 은 그대로
--         남아 "이전된 펫의 글을 옛 보호자가 계속 올릴 수 있음" 식의 비현실적 동작.
--   해결: 분양 완료 = 가구가 바뀌는 것 → 기존 보호자 전원 제거 후 입양자만 owner.
--
-- [E] owner 가 자기 자신을 pet_guardians 에서 제거하는 경로 차단
--   문제: 부분 unique(펫당 owner 1명)는 두 owner 동시 존재만 막지, owner 가 자신을
--         삭제해 "owner 부재 + primary_guardian_id 고아" 상태가 되는 건 못 막음.
-- ============================================================================

-- ---------------------------------------------------------------------------
-- [A-1] 펫 점유 충돌 검사 함수 (appointments INSERT 백스톱)
-- ---------------------------------------------------------------------------
create or replace function app.tg_appointments_pet_busy_check()
returns trigger
language plpgsql
security definer
set search_path = ''
as $$
declare v_conflict int;
begin
  if new.status = 'scheduled' then
    select count(*) into v_conflict
      from public.appointments a2
      join public.post_pets pp2 on pp2.post_id = a2.post_id
     where a2.status = 'scheduled'
       and a2.id <> new.id                                     -- 자기 자신 제외
       and pp2.pet_id in (
         select pet_id from public.post_pets where post_id = new.post_id
       );
    if v_conflict > 0 then
      raise exception '이미 다른 약속이 진행 중인 반려동물이 게시글에 포함되어 있습니다';
    end if;
  end if;
  return new;
end;
$$;

create or replace trigger trg_appointments_pet_busy
  before insert on public.appointments
  for each row execute function app.tg_appointments_pet_busy_check();

-- ---------------------------------------------------------------------------
-- [A-2] applications 수락 트리거에 사전 검사 추가 (명확한 에러 + 게시글 상태 보존)
--       기존 흐름: 게시글 잠금 → recruiting 확인 → matched 갱신 → appointments INSERT
--       새 흐름:   게시글 잠금 → recruiting 확인 → 펫 점유 사전 검사 → matched 갱신 → INSERT
-- ---------------------------------------------------------------------------
create or replace function app.tg_applications_on_accept()
returns trigger
language plpgsql
security definer
set search_path = ''
as $$
declare
  v_post      public.posts%rowtype;
  v_locked_id uuid;
  v_conflict  int;
begin
  if not (old.status = 'pending' and new.status = 'accepted') then
    return new;
  end if;

  -- 게시글 row lock (동시 수락 직렬화)
  select * into v_post from public.posts where id = new.post_id for update;
  if v_post.id is null then
    raise exception 'applications: 게시글이 존재하지 않습니다';
  end if;

  -- [A] 펫 점유 사전 검사: 게시글에 연결된 펫 중 이미 scheduled 약속이 있는 게 있나
  select count(*) into v_conflict
    from public.appointments a2
    join public.post_pets pp2 on pp2.post_id = a2.post_id
   where a2.status = 'scheduled'
     and pp2.pet_id in (
       select pet_id from public.post_pets where post_id = new.post_id
     );
  if v_conflict > 0 then
    raise exception '이 반려동물은 이미 다른 약속이 진행 중입니다. 해당 약속을 완료/취소한 뒤 수락해주세요';
  end if;

  -- recruiting → matched (한 명만 성공)
  update public.posts
     set progress_status = 'matched'
   where id = new.post_id and progress_status = 'recruiting'
  returning id into v_locked_id;
  if v_locked_id is null then
    raise exception '다른 사용자가 먼저 수락하였습니다';
  end if;

  -- 약속 생성
  insert into public.appointments
    (application_id, post_id, post_owner_id, applicant_id, status, scheduled_at)
  values
    (new.id, new.post_id, v_post.user_id, new.applicant_id, 'scheduled', v_post.scheduled_at);

  return new;
end;
$$;

-- ---------------------------------------------------------------------------
-- [B][C] applications BEFORE INSERT: 보호자 자기지원 + 비활성 펫 게시글 지원 차단
--        (0004 / 0008 흐름 유지 + 검사 추가)
-- ---------------------------------------------------------------------------
create or replace function app.tg_applications_block_insert()
returns trigger
language plpgsql
security definer
set search_path = ''
as $$
declare
  v_owner uuid;
  v_prog  text;
  v_vis   text;
begin
  select user_id, progress_status, visibility_status
    into v_owner, v_prog, v_vis
    from public.posts where id = new.post_id;

  if v_owner is null then
    raise exception 'applications: 존재하지 않는 게시글';
  end if;
  if v_owner = new.applicant_id then
    raise exception 'applications: 본인 게시글에는 지원할 수 없습니다';
  end if;
  if v_vis like 'deleted_%' then
    raise exception 'applications: 삭제된 게시글에는 지원할 수 없습니다';
  end if;
  if v_prog <> 'recruiting' then
    raise exception 'applications: 모집이 마감된 게시글입니다 (progress=%)', v_prog;
  end if;

  -- [B] 신청자가 게시글 펫의 보호자(owner/co_guardian)면 차단
  if exists (
    select 1 from public.post_pets pp
      join public.pet_guardians g on g.pet_id = pp.pet_id
     where pp.post_id = new.post_id and g.user_id = new.applicant_id
  ) then
    raise exception 'applications: 본인이 보호 중인 반려동물의 게시글에는 지원할 수 없습니다';
  end if;

  -- [C] 게시글에 비활성(active 아님) 펫이 포함되어 있으면 신규 지원 차단
  if exists (
    select 1 from public.post_pets pp
      join public.pets p on p.id = pp.pet_id
     where pp.post_id = new.post_id and p.pet_status <> 'active'
  ) then
    raise exception 'applications: 비활성 반려동물이 포함된 게시글에는 지원할 수 없습니다';
  end if;

  return new;
end;
$$;

-- ---------------------------------------------------------------------------
-- [D] 분양 완료 시: 기존 보호자 전원 정리 + 입양자만 owner 로
--     (0008 의 tg_appointments_after_update 재정의)
-- ---------------------------------------------------------------------------
create or replace function app.tg_appointments_after_update()
returns trigger
language plpgsql
security definer
set search_path = ''
as $$
declare
  v_category text;
  v_pet      uuid;
begin
  if old.status = 'scheduled' and new.status = 'completed' then
    update public.posts
       set progress_status = 'completed'
     where id = new.post_id and progress_status = 'matched';
    update public.applications
       set status = 'completed'
     where id = new.application_id and status = 'accepted';

    -- 분양 완료 → 가구 자체가 바뀌므로 기존 보호자 전원 정리, 입양자만 owner
    select category into v_category from public.posts where id = new.post_id;
    if v_category = 'give_away' then
      select pet_id into v_pet from public.post_pets where post_id = new.post_id limit 1;
      if v_pet is not null then
        delete from public.pet_guardians where pet_id = v_pet;
        insert into public.pet_guardians (pet_id, user_id, role, invited_by)
        values (v_pet, new.applicant_id, 'owner', new.post_owner_id);
        update public.pets set primary_guardian_id = new.applicant_id where id = v_pet;
      end if;
    end if;

  elsif old.status = 'scheduled' and new.status = 'cancelled' then
    update public.posts
       set progress_status = 'recruiting'
     where id = new.post_id and progress_status = 'matched';
    update public.applications
       set status = 'cancelled'
     where id = new.application_id and status = 'accepted';
  end if;

  return new;
end;
$$;

-- ---------------------------------------------------------------------------
-- [E] owner 가 자신을 pet_guardians 에서 제거하는 것 차단
--     (입양자 자동 승격 트리거가 자신을 옛 owner 자리에서 빼는 케이스는 허용)
-- ---------------------------------------------------------------------------
create or replace function app.tg_pet_guardians_prevent_owner_self_remove()
returns trigger
language plpgsql
security definer
set search_path = ''
as $$
begin
  -- 시스템(definer 분양 이전 등)은 app.uid()=NULL 또는 다른 컨텍스트라 우회 가능.
  -- 사용자 호출 컨텍스트(app.uid() is not null)에서만 강제.
  if old.role = 'owner' and app.uid() is not null and old.user_id = app.uid() then
    raise exception 'pet_guardians: owner 본인은 직접 제거할 수 없습니다. 먼저 소유권을 다른 보호자에게 이전하세요';
  end if;
  return old;
end;
$$;

create or replace trigger trg_pet_guardians_owner_self_remove
  before delete on public.pet_guardians
  for each row execute function app.tg_pet_guardians_prevent_owner_self_remove();
```

## 주의사항

- self-remove 차단은 시스템 정의자 경로(app.uid() NULL)에선 통과 → 분양 이전 트리거는 정상 동작.
- 펫 점유 검사는 pet 이 게시글의 post_pets 에 있는지 기준. 입양(adoption) 게시글은 post_pets 없음 → 0013 에서 offered_pet_id 로 확장.
