-- ============================================================================
-- PawMate · 0016 · post_likes → post_hearts 리네임 (명칭 통일)
-- ----------------------------------------------------------------------------
-- 배경: 요구사항 명세서 F-27 의 "하트(관심 게시글)" 기능을 0002 에서 `post_likes`
--   라는 이름으로 만들면서, 도메인상 별개 개념인 "좋아요(=초기 단계에 제거되어
--   평가 시스템으로 흡수)" 와 이름이 겹쳐 혼동을 유발. 별도 좋아요 테이블·트리거는
--   처음부터 존재하지 않았고 본 마이그레이션은 명칭만 하트로 통일.
-- 변경: 테이블/컬럼/함수/트리거/인덱스/제약/정책/코멘트 일괄 리네임 (재실행 안전).
-- 앱 영향: PostgREST 엔드포인트가 /post_likes → /post_hearts 로, 컬럼이
--   like_count → heart_count 로 바뀜. Flutter 쪽 쿼리 참조 업데이트 필요.
-- ============================================================================

-- ---------------------------------------------------------------------------
-- 1) 테이블 리네임 (재실행 안전)
-- ---------------------------------------------------------------------------
do $$
begin
  if exists (select 1 from information_schema.tables
              where table_schema='public' and table_name='post_likes')
     and not exists (select 1 from information_schema.tables
              where table_schema='public' and table_name='post_hearts') then
    execute 'alter table public.post_likes rename to post_hearts';
  end if;
end $$;

-- ---------------------------------------------------------------------------
-- 2) posts.like_count → posts.heart_count
-- ---------------------------------------------------------------------------
do $$
begin
  if exists (select 1 from information_schema.columns
              where table_schema='public' and table_name='posts' and column_name='like_count')
     and not exists (select 1 from information_schema.columns
              where table_schema='public' and table_name='posts' and column_name='heart_count') then
    execute 'alter table public.posts rename column like_count to heart_count';
  end if;
end $$;

-- ---------------------------------------------------------------------------
-- 3) 트리거 함수 본문 갱신 (heart_count 참조) + 함수명 변경
--    CREATE OR REPLACE 로 본문 갱신 → ALTER FUNCTION 으로 이름만 교체.
--    트리거는 함수를 OID 로 참조하므로 자동 따라감.
-- ---------------------------------------------------------------------------
create or replace function app.tg_post_likes_count()
returns trigger
language plpgsql
security definer
set search_path = ''
as $$
begin
  if tg_op = 'INSERT' then
    update public.posts set heart_count = heart_count + 1 where id = new.post_id;
    return new;
  elsif tg_op = 'DELETE' then
    update public.posts set heart_count = greatest(heart_count - 1, 0) where id = old.post_id;
    return old;
  end if;
  return null;
end;
$$;

do $$
begin
  if exists (select 1 from pg_proc p
              join pg_namespace n on n.oid = p.pronamespace
             where n.nspname='app' and p.proname='tg_post_likes_count')
     and not exists (select 1 from pg_proc p
              join pg_namespace n on n.oid = p.pronamespace
             where n.nspname='app' and p.proname='tg_post_hearts_count') then
    execute 'alter function app.tg_post_likes_count() rename to tg_post_hearts_count';
  end if;
end $$;

-- ---------------------------------------------------------------------------
-- 4) 트리거명 변경
-- ---------------------------------------------------------------------------
do $$
begin
  if exists (select 1 from pg_trigger
              where tgname='trg_post_likes_count'
                and tgrelid = 'public.post_hearts'::regclass) then
    execute 'alter trigger trg_post_likes_count on public.post_hearts rename to trg_post_hearts_count';
  end if;
end $$;

-- ---------------------------------------------------------------------------
-- 5) 인덱스 / 제약 / 정책 이름 정리
-- ---------------------------------------------------------------------------
do $$
begin
  if exists (select 1 from pg_class where relname='post_likes_user_idx')
     and not exists (select 1 from pg_class where relname='post_hearts_user_idx') then
    execute 'alter index public.post_likes_user_idx rename to post_hearts_user_idx';
  end if;
end $$;

do $$
begin
  if exists (select 1 from pg_constraint
              where conname='post_likes_uq'
                and conrelid = 'public.post_hearts'::regclass) then
    execute 'alter table public.post_hearts rename constraint post_likes_uq to post_hearts_uq';
  end if;
end $$;

-- 정책은 drop+create 가 가장 안전 (이름 변경 ALTER 는 안 됨)
drop policy if exists post_likes_select on public.post_hearts;
drop policy if exists post_likes_insert on public.post_hearts;
drop policy if exists post_likes_delete on public.post_hearts;
drop policy if exists post_hearts_select on public.post_hearts;
drop policy if exists post_hearts_insert on public.post_hearts;
drop policy if exists post_hearts_delete on public.post_hearts;

create policy post_hearts_select on public.post_hearts
  for select using (true);
create policy post_hearts_insert on public.post_hearts
  for insert with check (user_id = app.uid());
create policy post_hearts_delete on public.post_hearts
  for delete using (user_id = app.uid());

-- ---------------------------------------------------------------------------
-- 6) 코멘트 갱신 — 좋아요 표기 제거, 하트 기능임을 명확히
-- ---------------------------------------------------------------------------
comment on table public.post_hearts is
  '하트(관심 게시글/북마크). 요구사항 F-27. 토글 = 하드 DELETE(soft delete 사용 안 함). UNIQUE(post_id, user_id) → 재생성 자유. ※ 좋아요(like) 와는 별개 개념 — 좋아요는 초기 단계에 제거되어 평가 시스템으로 흡수됨';
comment on column public.posts.heart_count is
  '하트 누적 캐시. tg_post_hearts_count 트리거가 INSERT/DELETE 시 ±1 자동 동기화';
