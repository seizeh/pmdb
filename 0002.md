# 0002 · 테이블 및 제약

**파일**: `supabase/migrations/20260525120002_tables.sql`
**카테고리**: 초기 세트업
**의존성**: 0001

## 목적

PawMate 전체 스키마의 26개 테이블·모든 컬럼·CHECK 제약·FK 관계·순환 FK(chat_rooms↔chat_messages) 해소를 담당합니다. Soft-delete 우선 철학에 따라 이력성 참조는 NO ACTION, 순수 junction만 CASCADE로 설계.

| 알림 | notifications, device_tokens, notification_preferences |

-- ▸ reviews  (completed 약속 당사자만, 1인 1회, 불변)
--   카테고리 정규형(no-space, 요구사항 F-25 기준):
--     긍정: 친절해요 / 약속을잘지켜요 / 반려동물이순해요 / 준비성이좋아요
--     부정: 불친절해요 / 약속을잘안지켜요 / 반려동물이사나워요 / 준비성이아쉬워요
create table public.reviews (
  id             uuid primary key default gen_random_uuid(),
  appointment_id uuid not null references public.appointments(id),
  reviewer_id    uuid not null references public.users(id),
  reviewee_id    uuid not null references public.users(id),
  categories     text[] not null,
  created_at     timestamptz not null default now(),
  constraint reviews_uq unique (appointment_id, reviewer_id),
  constraint reviews_self_chk check (reviewer_id <> reviewee_id),
  constraint reviews_len_chk  check (array_length(categories, 1) between 1 and 4),
  -- 허용 집합 외 값 차단
  constraint reviews_allowed_chk check (
    categories <@ array[
      '친절해요','약속을잘지켜요','반려동물이순해요','준비성이좋아요',
      '불친절해요','약속을잘안지켜요','반려동물이사나워요','준비성이아쉬워요'
    ]::text[]
  ),
  -- 상호 배제 4쌍
  constraint reviews_excl_kind check (
    not ('친절해요' = any(categories) and '불친절해요' = any(categories))),
  constraint reviews_excl_promise check (
    not ('약속을잘지켜요' = any(categories) and '약속을잘안지켜요' = any(categories))),
  constraint reviews_excl_temper check (
    not ('반려동물이순해요' = any(categories) and '반려동물이사나워요' = any(categories))),
  constraint reviews_excl_prepared check (
    not ('준비성이좋아요' = any(categories) and '준비성이아쉬워요' = any(categories)))
);

create table public.location_verifications (
  id                     uuid primary key default gen_random_uuid(),

                           check (result in ('success','failed','blocked')),

-- ▸ user_blocks  (단방향 차단)
create table public.user_blocks (
  id         uuid primary key default gen_random_uuid(),

  blocked_id uuid not null references public.users(id),
  created_at timestamptz not null default now(),
  constraint user_blocks_uq unique (blocker_id, blocked_id),
  constraint user_blocks_self_chk check (blocker_id <> blocked_id)

-- ▸ facility_cache  (지도 시설물 캐시)
create table public.facility_cache (
  id                  uuid primary key default gen_random_uuid(),
  kakao_place_id      varchar(50) not null,
  source_provider     varchar(20) not null default 'kakao'
                        check (source_provider in ('kakao','naver','google')),
  name                varchar(100) not null,
  category            varchar(30) not null,
  address             text,

  lng                 decimal(10,7) not null,
  phone               varchar(20),
  website_url         text,
  business_hours      jsonb,
  thumbnail_url       text,
  is_open_now         boolean,
  open_status_updated_at timestamptz,
  cached_at           timestamptz not null default now(),
  expires_at          timestamptz not null,
  last_api_sync_at    timestamptz,
  sync_fail_count     smallint not null default 0 check (sync_fail_count >= 0),

## 주의사항

- 순환 FK 해소: chat_rooms.last_message_id → chat_messages.id 는 chat_messages 생성 후 ALTER 로 추가.
- reviews.categories 는 TEXT[] 배열. 상호 배제 4쌍을 CHECK 로 강제(친절해요/불친절해요 등).
- reports.target_type + target_id 는 polymorphic — FK 걸지 않음. 서비스 레이어 검증.
- posts 는 visibility_status + progress_status 두 축 독립 관리 (single-source 는 is_deleted 아님).
