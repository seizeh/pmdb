# 0015 · 푸시 알림 파이프라인 (FCM/APNs)

**파일**: `supabase/migrations/20260525120015_push_pipeline.sql`
**카테고리**: 인프라
**의존성**: 0004

## 목적

인앱(Realtime, 포그라운드)과 푸시(Edge Function → FCM/APNs, 백그라운드) 채널을 분리. Realtime 만으로는 백그라운드/종료 상태 알림이 불가하므로 Database Webhook → Edge Function → FCM/APNs 파이프라인을 DB 측 상태머신과 RPC 로 지원.

## 핵심 변경

| 구분 | 내용 |
|---|---|
| notifications 컬럼 | push_status(pending|sent|failed|skipped), push_attempts, push_error |
| 펜딩 큐 인덱스 | notifications(created_at) WHERE push_status='pending' — 폴링 fallback |
| RPC | app.mark_push_sent, app.mark_push_failed(재시도·영구실패 분기), app.mark_push_skipped, app.deactivate_device_token |

## 전체 SQL

```sql
-- ============================================================================
-- PawMate · 0015 · 푸시 알림 파이프라인 (FCM/APNs 발송 레이어)
-- ----------------------------------------------------------------------------
-- 알림은 두 채널을 분리해서 다룹니다:
--   ① 인앱(Realtime, WebSocket) — 포그라운드 UI 갱신(뱃지·미읽음·새 메시지 표시).
--      이 경로는 0004 의 unread_count 트리거와 클라이언트 Realtime 구독이 담당.
--   ② 푸시(FCM/APNs) — 백그라운드/종료 상태 알림. Realtime 으로는 못 닿는 영역.
--      권장 흐름:
--         INSERT into notifications (push_status='pending')
--           → Supabase Database Webhook 이 Edge Function 호출
--           → Edge Function 이 notification_preferences/device_tokens 조회
--           → FCM/APNs 로 발송 (silent push 는 is_silent=true 일 때 data-only)
--           → 결과를 app.mark_push_sent() / app.mark_push_failed() 로 갱신
--           → InvalidToken/UNREGISTERED 응답은 app.deactivate_device_token() 으로 처리
--
-- 본 마이그레이션은 DB 측 파이프라인(상태머신/RPC/큐 인덱스)을 정돈합니다.
-- Edge Function 코드 자체는 supabase/functions/ 아래 TypeScript 로 별도 작성.
-- ============================================================================

-- ---------------------------------------------------------------------------
-- 1) 푸시 상태 머신
-- ---------------------------------------------------------------------------
alter table public.notifications
  add column if not exists push_status varchar(20) not null default 'pending'
    check (push_status in ('pending','sent','failed','skipped')),
  add column if not exists push_attempts smallint not null default 0
    check (push_attempts >= 0),
  add column if not exists push_error text;

comment on column public.notifications.push_status is
  '푸시 발송 상태. pending=미발송/재시도대기, sent=성공, failed=영구실패(임계 초과), skipped=조건상 미발송(사용자 설정/토큰없음 등). 권장 흐름: Database Webhook → Edge Function 이 갱신';
comment on column public.notifications.push_attempts is
  '푸시 발송 시도 횟수. mark_push_failed 호출 시 임계치(기본 3회) 초과하면 push_status=failed 로 자동 전환';
comment on column public.notifications.push_error is
  '마지막 푸시 실패 사유 텍스트(디버그/모니터링용)';

-- 기존 push_sent / push_sent_at 컬럼은 백워드 호환을 위해 유지하되,
-- mark_push_sent 가 함께 갱신하므로 사실상 push_status='sent' 의 동의어로 동작.

-- ---------------------------------------------------------------------------
-- 2) 펜딩 푸시 큐 인덱스 (Webhook 실패 시 폴링/재시도 fallback 용)
-- ---------------------------------------------------------------------------
create index if not exists notifications_push_pending_idx
  on public.notifications (created_at)
  where push_status = 'pending';

-- ---------------------------------------------------------------------------
-- 3) Edge Function 이 호출할 RPC 들
-- ---------------------------------------------------------------------------

-- 3-1) 발송 성공 기록
create or replace function app.mark_push_sent(p_notification_id uuid)
returns void
language sql
security definer
set search_path = ''
as $$
  update public.notifications
     set push_status  = 'sent',
         push_sent    = true,
         push_sent_at = now(),
         updated_at   = now()
   where id = p_notification_id
$$;
grant execute on function app.mark_push_sent(uuid) to service_role;
comment on function app.mark_push_sent(uuid) is
  'Edge Function 이 FCM/APNs 성공 응답 시 호출. push_status=sent + push_sent_at 갱신';

-- 3-2) 발송 실패 기록 (재시도 가능 / 영구실패 분기)
--      p_final=true 면 즉시 failed. 아니면 push_attempts 증가 후 임계 초과 시 failed.
create or replace function app.mark_push_failed(
  p_notification_id uuid,
  p_error           text default null,
  p_final           boolean default false,
  p_max_attempts    smallint default 3
)
returns void
language plpgsql
security definer
set search_path = ''
as $$
declare
  v_current smallint;
  v_next    smallint;
begin
  select push_attempts into v_current
    from public.notifications where id = p_notification_id;
  if v_current is null then
    raise exception 'mark_push_failed: 알림(%) 을 찾을 수 없음', p_notification_id;
  end if;
  v_next := v_current + 1;

  update public.notifications
     set push_attempts = v_next,
         push_error    = p_error,
         push_status   = case
           when p_final or v_next >= p_max_attempts then 'failed'
           else 'pending'   -- 다음 webhook/폴링이 다시 가져감
         end,
         updated_at    = now()
   where id = p_notification_id;
end;
$$;
grant execute on function app.mark_push_failed(uuid, text, boolean, smallint) to service_role;
comment on function app.mark_push_failed(uuid, text, boolean, smallint) is
  'Edge Function 이 FCM/APNs 실패 응답 시 호출. p_final=true 또는 시도횟수 임계 초과 시 영구실패(failed), 아니면 pending 유지로 재시도 큐 잔류';

-- 3-3) 발송 건너뜀 기록 (preferences off / device_token 없음 등)
create or replace function app.mark_push_skipped(p_notification_id uuid, p_reason text default null)
returns void
language sql
security definer
set search_path = ''
as $$
  update public.notifications
     set push_status = 'skipped',
         push_error  = p_reason,
         updated_at  = now()
   where id = p_notification_id
$$;
grant execute on function app.mark_push_skipped(uuid, text) to service_role;

-- 3-4) 유효하지 않은 디바이스 토큰 비활성화 (InvalidToken/UNREGISTERED 응답 처리)
create or replace function app.deactivate_device_token(p_token text, p_reason text default null)
returns void
language sql
security definer
set search_path = ''
as $$
  update public.device_tokens
     set is_active     = false,
         failure_count = failure_count + 1,
         updated_at    = now()
   where token = p_token
$$;
grant execute on function app.deactivate_device_token(text, text) to service_role;
comment on function app.deactivate_device_token(text, text) is
  'Edge Function 이 APNs InvalidToken / FCM UNREGISTERED 응답 시 호출. is_active=false 로 발송 대상에서 제외';
```

## 주의사항

- 인앱 UI 갱신은 여전히 Realtime + 0004 트리거로 처리.
- silent push(is_silent=true)는 여전히 발송되나 FCM data-only / APNs content-available:1.
- Edge Function TypeScript 는 supabase/functions/ 아래 별도.
